Animation

 
-- Animation(Element this._element, num this._duration, Map properties, AnimationOptions this._options)

final Animation anim = new Animation(query('#box05'), 3,
        {'opacity': {'from':1, 'to':0},
         'width': '150px',
         'margin-left': {'from':'0px', 'to':'25px'},
         'background-position':'-20px -20px'}, new AnimationOptions());
anim.start();
